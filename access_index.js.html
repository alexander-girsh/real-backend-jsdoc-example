const crypto = require('crypto')
const {config} = require('./config.js')
const JWT = require('jsonwebtoken')
const {error} = require('../logger/index.js')
const {codes} = require('status-codes')

/**
 * @module access
 * @desc It exports methods to work with content of authorization cookies
 * This methods set was extracted cause multiply services requires it
 */



/** creating a reusable crypto-module-friendly key using secrets from .env
 * I call this operation once cause it takes 50ms
 * */
/**
 * @desc Secret key to use encoding / decoding the authentication token
 * @see {@link module:http-server/config.AUTH_TOKEN_ENCRYPTION_SALT}
 * @member ENCRYPTION_KEY
 * @type {Buffer}
 */
const ENCRYPTION_KEY = crypto.scryptSync(config.AUTH_TOKEN_ENCRYPTION_SECRET, config.AUTH_TOKEN_ENCRYPTION_SALT,  24)
const IV_MOCK = ''


/**
 * It encrypts the provided .data_string using algo des-ede3 and native "crypto" node module
 * @param data_string {String} plain string
 * @returns {String} des-ede3 encrypted string
 */
const encryptString = (data_string) => {


    const cipher = crypto.createCipheriv('des-ede3', ENCRYPTION_KEY, IV_MOCK)

    return cipher.update(data_string, 'utf8', 'hex') + cipher.final('hex')
}

/**
 * It decrypts data encrypted earlier using the .encryptString() method
 * If variables config.AUTH_TOKEN_ENCRYPTION_SECRET or config.AUTH_TOKEN_ENCRYPTION_SALT were updated,
 * between the encryption and decryption, te earlier encrypted string may not be decrypted.
 * @param encrypted_data_string {String} des-ede3 encrypted string
 * @returns {String} decrypted plain string
 */
module.exports.decryptString = (encrypted_data_string) => {

    try {

        const decipheriv = crypto.createDecipheriv('des-ede3', ENCRYPTION_KEY, IV_MOCK)

        return decipheriv.update(encrypted_data_string, 'hex', 'utf8') + decipheriv.final('utf8')

    } catch (e) {
        return null
    }

}

/**
 * It packs the provided user data to the JWT token signed using {@link module:http-server/config.JWT_SECRET}
 * @function issueJWTToken
 * @param user_data {AuthTokenContent} user data to pack into the token to get fast access handling requests
 * authorized with this token
 * @returns {Promise<String>}
 */
module.exports.issueJWTToken = async (user_data) => {

    try {

        const data_to_encrypt = config.ALLOWED_JWT_FIELDS.reduce((data, field_name) => {
            if (user_data[field_name] !== undefined) {
                data[field_name] = user_data[field_name]
            }

            return data

        }, {})

        /** this jwt is not encrypted and can be viewed using for e.x. https://jwt.io,
         * but the payload into it can not be changed by user
         */
        const jwt = await JWT.sign(data_to_encrypt, config.JWT_SECRET)

        /**
         * It returns the hard-ciphered jwt and user can not decode it independently and see the data.
         * I am sorry but the ciphered jwt is much longer that the pure jwt and it will generate some
         * overhead to http-server-traffic. If you know a better algo with shorten result then 'des-ede3'
         * let me know. Some good news - I am sure that we have no encryption CPU / MEM overhead.
         */
        return encryptString(jwt)


    } catch (e) {
        error('access/issueJWTToken err:', e)
    }

}


module.exports.decodeJWTToken = async (token) => {

    try {

        /** on verification fail it will throw an exception
         * @type AuthTokenContent
         * */
        const token_content = await JWT.verify(token, config.JWT_SECRET)

        /** adding data from jwt to the request object
         * next middlewares/endpoints will be able to process this data
         */

        if (token_content) {
            /**
             * @type {AuthTokenContent}
             */
            // eslint-disable-next-line require-atomic-updates
            return {status: codes.OK, token_content}

        } else {
            return {status: codes.FAIL}
        }




    } catch (e) {

        /** doing nothing cause JWT.verify( ) will throw exceptions if token is not verified */

        return {status: codes.FAIL}
    }
}

module.exports.hasAdminPermissions = (permissions) => {

    return permissions === config.PERMISSIONS.ADMIN
}

module.exports.hasUserPermissions = (permissions) => {

    return permissions === config.PERMISSIONS.USER
}

module.exports.USER_PERMISSIONS = config.PERMISSIONS.USER
module.exports.ADMIN_PERMISSIONS = config.PERMISSIONS.ADMIN
module.exports.COOKIES_SECRET = config.COOKIES_SECRET
