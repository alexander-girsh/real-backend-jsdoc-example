database/connection.js

/**
 * @section connection
 * @module database/connection
 * @desc Exports methods to use the main database (usually a cloud-based Postgres)
 */

const PG = require( 'pg')
const {setTypeParsers} = require( './types.js')
const createSubscriber = require( 'pg-listen')
const {info, error, fatal} = require( '../logger/index.js')
const {config} = require( './config.js')
const {codes} = require( 'status-codes')



setTypeParsers(PG)


const PG_POOL_CONFIG = {
    ssl: process.env.NODE_ENV === 'production' ? {
        rejectUnauthorized: false
    } : false,
    max: 400 // todo не освобождает клиентов после транзакции!!!
}



const pool = new PG.Pool(PG_POOL_CONFIG)

pool.on('error', (err) => {

    error({err}, `Unexpected error of database connection`)

    reconnect()
})

/**
 * Adds onError handler for each connected client
 * @param client {PG.Client}
 */
const onClientConnected = client => {
    pool.emit('connect')
    info(`DB client connected to ${process.env.PGHOST}`)
    client.on('error', err => {
        error({err}, `DB client disconnected, trying to reconnect`)
        if (err.code === 'ECONNRESET' || err.code === 'EPIPE') reconnect()
    })}


/**
 * Tries to reconnect to database. If the number of attempts is exhausted,
 * terminates the process with exit code 1
 */
const reconnect = () => {
    if (!pool._reconnecting) {
        pool._reconnecting = true
        let currentAttempt = 0
        const reconnectionInterval = setInterval(() => {
            if (currentAttempt === config.RECONNECTION_ATTEMPTS) {
                fatal( 'Can not reconnect to DB')
                clearInterval(reconnectionInterval)
                pool._reconnecting = false
                process.exit(1)
            }
            currentAttempt++
            info('Reconnection attempt #' + currentAttempt)
            pool.connect()
                .then(client => {
                    onClientConnected(client)
                    pool._reconnecting = false
                    clearInterval(reconnectionInterval)
                })
                .catch(err => pool.emit('error', err))
        }, config.RECONNECTION_DELAY)
    }
}

const connect = pool => {
    return pool.connect()
        .then(onClientConnected)
        .catch(err => {
            pool.emit('error', err)
            reconnect()
        })
}

if (process.env.NODE_ENV !== 'test') {
    connect(pool)
}


const addSubscriber = async (subscriberPayload) => {
    try {
        const subscriber = await createSubscriber({
            connectionString: config.DB_CONNECTION_URL,
            ssl: {
                rejectUnauthorized: false
            }
        })
        await subscriber.connect()

        return await subscriberPayload(subscriber)

    } catch (err) {
        error({err}, `addSubscriber() err:`, err)
    }
}




/**
 * @return {Promise<PoolClient>}
 */

const createNewClient = async () => {

    const client = await pool.connect()


    info({
        labels: ['pg-pool'],
        request: 'createNewClient',
        status: 'SUCCESS',
        pool_state: {
            total_count: pool.totalCount,
            idle_count: pool.idleCount,
            waiting_count: pool.waitingCount
        }
    })

    return client
}

/**
 * @param {Function} payload - an async function which takes
 * @param {Object} options
 * @param {String} options.ISOLATION_LEVEL - pg transaction isolation level, for e.g and for default is SERIALIZABLE"
 * @param {Number} [options.retries_limit]
 * @returns {Promise<{status: *}>}
 */
const execRepeatableTransaction = async (payload, {ISOLATION_LEVEL = 'READ COMMITTED', retries_limit = config.DB_TRANSACTIONS_RETRIES_LIMIT} = {}) => {

    let client


    try {

        if (!['READ COMMITTED', 'REPEATABLE READ', 'SERIALIZABLE'].includes(ISOLATION_LEVEL)) {
            return {status: codes.WRONG_ARGS}
        }

        client = await pool.connect()

        await client.query('BEGIN')


        for (let retries_count = 0; retries_count < retries_limit; retries_count++) {

            try {

                await client.query('START TRANSACTION')

                await client.query(`SET TRANSACTION ISOLATION LEVEL ${ISOLATION_LEVEL}`)


                const result = await payload(client, retries_count)


                if (codes.statusEqual(result.status, codes.FAIL)) {
                    await client.query('ROLLBACK')
                    continue
                }

                if (!codes.statusEqual(result.status, codes.OK)) {
                    await client.query('ROLLBACK')
                    return result
                }

                await client.query('COMMIT')

                return result

            } catch (err) {
                if (err.routine === 'PreCommit_CheckForSerializationFailure') {
                    continue
                }
                error('Error executing the transaction:', err)
                return {status: codes.FAIL, error: err}
            }

        }

        return {status: codes.IMPOSSIBLE}


    } catch (err) {
        error('database/execRepeatableTransaction err:', err)
        return {status: codes.FAIL}
    } finally {

        if (client) {
            await client.query('END')
        }

        await releaseClientSilly(client)
    }

}


async function releaseClientSilly (client) {
    try {

        await client.release()

    } catch (e) {
        if (e.message !== 'Release called on client which has already been released to the pool.') {
            error(`releaseClientSilly() err:`, e)
            throw e
        }
        /** doing nothing */
    }
}

const DB = {
    query: async (...args) => {

        let query_start_timestamp
        let query_result

        try {

            /** adding new connection to pool if there is no free connections */
            if (!pool.idleCount || pool.waitingCount) {
                const client = await pool.connect()
                await releaseClientSilly(client)
            }


            query_start_timestamp = Date.now()

            return await pool.query(...args)

        } catch (e) {
            throw e
        }  finally {

            const query_end_timestamp = Date.now()

            if (config.PG_POOL_QUERIES_VERBOSE_MODE) info({
                labels: ['pg-pool'],
                query: args[0],
                query_result_status: query_result instanceof Error ? 'ERROR' : 'SUCCESS',
                time_spent_ms: query_end_timestamp - query_start_timestamp,
                pool_state: {
                    total_count: pool.totalCount,
                    idle_count: pool.idleCount,
                    waiting_count: pool.waitingCount
                }
            })
        }
    }
}
