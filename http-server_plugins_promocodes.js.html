const {codes} = require( 'status-codes')
const {DB, execRepeatableTransaction} = require( '../../database/connection.js')
const {error, info} = require( '../../logger/index.js')
const {nats} = require( 'nats-connection/index.cjs')
const {config} = require( '../config.js')


const publishPromocodeActivationUpdateEvent = ({user_id, promocode_details}) => {
    try {
        nats.publish('PERSONAL.PROMOCODES.ACTIVATION_UPDATE', {
            user_id, ...promocode_details
        })
    } catch (e) {
        console.error(`publishPromocodeActivationEvent() err:`, e)
        return {status: codes.FAIL}
    }
}

/** promocodes with 'INSTANT' bonus triggers are handled here */
const handleNewPromocodeActivation = async ({promocode_activation_id}) => {
    try {

        const {status, user_id, promocode_details} = await execRepeatableTransaction(async (client) => {

            const promocode_details = await client.query(`SELECT p.valid_before >= now() as promocode_is_still_valid, 
                                                                 p.*,
                                                                 pa.status,
                                                                 pa.user_id
            FROM promocodes_activations pa
                LEFT JOIN promocodes p ON (pa.promocode_text = p.promocode_text)
                WHERE pa.promocode_activation_id = $1
                `, [
                promocode_activation_id
            ])
                .then(r => r.rows[0])

            if (!promocode_details.promocode_is_still_valid) {
                return {status: codes.INCORRECT_STATE}
            }

            if (promocode_details.status !== 'CREATED') {
                return {status: codes.INCORRECT_STATE}
            }

            /** creating new bonus instantly */
            if (promocode_details.bonus_trigger === 'INSTANT') {

                if (promocode_details.bonus_kind === 'FIXED') {

                    const bonus_amount = parseFloat(promocode_details.fixed_bonus_usd_amount.toFixed(2))

                    const bonus = await client.query(`INSERT INTO promocodes_bonuses (promocode_activation_id, bonus_amount) VALUES ($1, $2) RETURNING *`, [
                        promocode_activation_id,
                        bonus_amount
                    ])
                        .then(r => r.rows[0])

                    if (!bonus) {
                        throw new Error(`Unable to create the promocode bonus for promocode activation ${JSON.stringify(promocode_details)}`)
                    }

                    const promocode_activation_updating_result = await client.query(`UPDATE promocodes_activations SET status = 'COMPLETED' 
                        WHERE promocode_activation_id = $1 AND status = 'CREATED'`, [
                        promocode_activation_id
                    ])

                    if (promocode_activation_updating_result.rowCount !== 1) {
                        return {status: codes.INCORRECT_STATE}
                    }

                    const {
                        created_at,
                        promocode_text,
                        valid_before,
                        bonus_kind,
                        bonus_trigger,
                        fixed_bonus_usd_amount,
                        percentage_bonus_value,
                        min_bonus_trigger_value
                    } = promocode_details

                    return {
                        status: codes.OK,
                        user_id: promocode_details.user_id,
                        promocode_details: {
                            created_at,
                            promocode_text,
                            valid_before,
                            bonus_kind,
                            bonus_trigger,
                            bonus_amount,
                            fixed_bonus_usd_amount,
                            percentage_bonus_value,
                            min_bonus_trigger_value,
                            promocode_activation_id,
                            status: 'COMPLETED'
                        }
                    }


                }

            }


            return {status: codes.OK}

        }, {ISOLATION_LEVEL: 'SERIALIZABLE'})

        if (!codes.statusEqual(status, codes.OK)) {
            return {status}
        }

        if (promocode_details && promocode_details.status === 'COMPLETED') {

            const updated_balance = await DB.query('SELECT calcBalance($1) as balance', [user_id]).then(r => r.rows[0].balance)

            nats.publish('PERSONAL.ACCOUNT.BALANCE_UPDATE', {user_id, balance: updated_balance})

            publishPromocodeActivationUpdateEvent({user_id, promocode_details})

        }

        return {status: codes.OK}

    } catch (e) {
        error(`handleNewPromocodeActivation() err:`, e)
        return {status: codes.FAIL}
    }
}


/** promocodes with 'DEPOSIT' bonus triggers are handled here */
const handleNewCompletedDeposit = async ({deposit_id, deposit_amount, user_id}) => {
    try {


        if (!deposit_id || !deposit_amount || !user_id) {
            return {status: codes.WRONG_ARGS}
        }

        /** seeking for activated promocode which requirements are fulfilled by amount of provided deposit */
        const matching_promocode_activation = await DB.query(`SELECT pa.created_at, 
            p.min_bonus_trigger_value,
            pa.promocode_activation_id
            FROM promocodes_activations pa
                LEFT JOIN promocodes p
                    ON (pa.promocode_text = p.promocode_text)
            WHERE pa.user_id = $1 
              AND pa.status = 'CREATED' 
                AND p.bonus_trigger = 'DEPOSIT'
                    AND p.min_bonus_trigger_value <= $2
                        AND p.valid_before >= now()`, [
            user_id,
            deposit_amount
        ])
            .then(r => r.rows[0])

        /** no matching promocode activation found */
        if (!matching_promocode_activation) {
            return {status: codes.NOT_FOUND}
        }

        const {status, promocode_details} = await execRepeatableTransaction(async (client) => {


            const deposit_details = await client.query(`SELECT d.net_amount_usd, deposit_id
                                                FROM deposits d
                                                         LEFT JOIN users u ON (d.user_id = u.user_id)
                                                WHERE d.deposit_id = $1 
                                                  AND d.status = 'COMPLETED'
                                                    AND d.created_at >= $2
                                                       AND d.net_amount_usd >= $3`,

            [
                deposit_id,
                matching_promocode_activation.created_at,
                matching_promocode_activation.min_bonus_trigger_value
            ])
                .then(r => r.rows[0])

            if (!deposit_details) {
                throw new Error(`Unable to find matching deposit: ${JSON.stringify({matching_promocode_activation, deposit_id, deposit_amount, user_id})}`)
            }



            const promocode_details = await client.query(`SELECT p.valid_before >= now() as promocode_is_still_valid, 
                                                                 p.*,
                                                                 pa.status,
                                                            
                                                                 pa.user_id
            FROM promocodes_activations pa
                LEFT JOIN promocodes p ON (pa.promocode_text = p.promocode_text)
                WHERE pa.promocode_activation_id = $1
                `, [
                matching_promocode_activation.promocode_activation_id
            ])
                .then(r => r.rows[0])


            if (!promocode_details) {
                throw new Error(`Unable to find matching deposit: ${JSON.stringify({matching_promocode_activation, deposit_id, deposit_details, user_id})}`)
            }


            const bonus_amount = (() => {
                if (promocode_details.bonus_kind === 'PERCENTAGE') {
                    return parseFloat((deposit_details.net_amount_usd * ( promocode_details.percentage_bonus_value / 100 )).toFixed(2))
                } else if (promocode_details.bonus_kind === 'FIXED') {
                    return parseFloat(promocode_details.fixed_bonus_usd_amount.toFixed(2))
                }   else {
                    return null
                }
            })()

            if (!bonus_amount) {
                throw new Error(`Unable to calculate bonus amount: ${JSON.stringify({deposit_details, promocode_details, matching_promocode_activation, user_id})}`)
            }


            const bonus = await client.query(`INSERT INTO promocodes_bonuses (promocode_activation_id, bonus_amount) VALUES ($1, $2) RETURNING *`, [
                matching_promocode_activation.promocode_activation_id,
                bonus_amount
            ])
                .then(r => r.rows[0])

            if (!bonus) {
                throw new Error(`Unable to create the promocode bonus for promocode activation ${JSON.stringify(promocode_details)}`)
            }

            const promocode_activation_updating_result = await client.query(`UPDATE promocodes_activations SET status = 'COMPLETED' 
                        WHERE promocode_activation_id = $1 AND status = 'CREATED'`, [
                matching_promocode_activation.promocode_activation_id
            ])

            if (promocode_activation_updating_result.rowCount !== 1) {
                return {status: codes.INCORRECT_STATE}
            }

            const {
                created_at,
                promocode_text,
                valid_before,
                bonus_kind,
                bonus_trigger,
                fixed_bonus_usd_amount,
                percentage_bonus_value,
                min_bonus_trigger_value
            } = promocode_details

            return {
                status: codes.OK,
                promocode_details: {
                    created_at,
                    promocode_text,
                    valid_before,
                    bonus_kind,
                    bonus_trigger,
                    bonus_amount,
                    fixed_bonus_usd_amount,
                    percentage_bonus_value,
                    min_bonus_trigger_value,
                    promocode_activation_id: matching_promocode_activation.promocode_activation_id,
                    status: 'COMPLETED'
                }
            }


        }, {ISOLATION_LEVEL: 'SERIALIZABLE'})


        if (!codes.statusEqual(status, codes.OK)) {
            return {status}
        }


        if (promocode_details && promocode_details.status === 'COMPLETED') {

            const updated_balance = await DB.query('SELECT calcBalance($1) as balance', [user_id]).then(r => r.rows[0].balance)

            nats.publish('PERSONAL.ACCOUNT.BALANCE_UPDATE', {user_id, balance: updated_balance})

            publishPromocodeActivationUpdateEvent({user_id, promocode_details})

        }

        return {status: codes.OK}




    } catch (e) {
        error('handleCompletedDeposit() err:', e)
        return {status: codes.FAIL}
    }
}
