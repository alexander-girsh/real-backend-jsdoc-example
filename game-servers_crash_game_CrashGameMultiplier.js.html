game-servers/crash/game/CrashGameMultiplier.js

const {DB} = require('../../../database/connection.js')
const {info, error} = require('../../../logger')
const {codes} = require('status-codes')

class CrashGameMultiplier {

    #multiplier
    #final_multiplier_id

    constructor (fixed_value, final_multiplier_id) {

        if (typeof fixed_value === 'number') {
            this.#multiplier = fixed_value
        }

        /** if the value from db passed */
        if (final_multiplier_id) {
            this.#final_multiplier_id = final_multiplier_id
        }

    }


    /** required to increment multiplier
     * @param {Number} game_already_on_for
     * */
    calcCurrentMultiplierByMilliseconds(game_already_on_for) {
        return 2 ** (  ((game_already_on_for / 1000) / 11) )
    }
    /**
     * @param { CrashGameMultiplier } max_allowed_multiplier
     * @param {Number} game_already_on_for
     */
    increment (max_allowed_multiplier, game_already_on_for) {

        const next_value = this.calcCurrentMultiplierByMilliseconds(game_already_on_for)

        const next_multiplier = new CrashGameMultiplier( next_value )

        if (next_multiplier.value > max_allowed_multiplier.value) {

            return new CrashGameMultiplier(max_allowed_multiplier.value)

        } else {

            return next_multiplier

        }

    }


    /**
     *
     * @param {CrashGameDBTransaction}  db_transaction
     * @returns {Promise<{final_multiplier: CrashGameMultiplier, status: Status}|{status: Status}>}
     */
    static async fromDB ({db_transaction}) {

        try {

            const random_final_multiplier_from_DB = await db_transaction.randomClient().query(`
                SELECT final_multiplier,
                       final_multiplier_id
                FROM activated_crash_game_final_multipliers
                ORDER BY random()
                LIMIT 1`)
                .then(r => r.rows[0])


            if (!random_final_multiplier_from_DB) {
                error(`Unable to read random final multiplier from db =(`)
                return {
                    status: codes.INCORRECT_STATE
                }
            }

            return {
                status: codes.OK,
                final_multiplier: new CrashGameMultiplier(
                    random_final_multiplier_from_DB.final_multiplier,
                    random_final_multiplier_from_DB.final_multiplier_id
                )
            }

        } catch (e) {
            error(`CrashGameFinalMultiplier.fromDB() err:`, e)
            return {status: codes.OK}
        }
    }


    get value () {
        return this.#multiplier
    }

    get toString () {
        return this.#multiplier.toFixed(2)
    }

    get to2NumbersPrecisionFloat () {
        return parseFloat(this.toString)
    }

    get finalMultiplierId () {
        return this.#final_multiplier_id || null
    }

}

module.exports = {CrashGameMultiplier}
