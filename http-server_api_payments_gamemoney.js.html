/**
 * @module http-server/api/payments/gamemoney

const Express = require( 'express')
const {codes} = require( 'status-codes')
const {error, info} = require( '../../../logger/index.js')
const {accessUser} = require( '../../middleware/access.js')
const {config} = require( '../../config.js')
const GameMoneyLib = require( '../../plugins/gamemoney-lib.js')
const {gamemoneyIPVerificationMiddleware, gamemoneyHookSignatureVerificationMiddleware} = require( '../../middleware/gamemoney.js')
const {DB} = require( '../../../database/connection.js')
const {handleNewCompletedDeposit} = require('../../plugins/promocodes.js')
const {nats} = require('nats-connection/index.cjs')


const router = Express.Router()




/**
 * It handles the deposits hooks from gamemoney backend
 * @chain {@link module:http-server/api/payments/gamemoney~gamemoneyHookSignatureVerificationMiddleware}
 * @chain This endpoint
 * @function handleDepositHook
 * @path {POST} /api/payments/gamemoney/deposit-hook
 * @code {200} if hook handled correctly
 * @code {424} if body doesn't contain the required params
 * @code {500} if internal error - hook should be repeated in future
 * @body {GamemoneyDepositHook}
 * @body received_amount
 * @response {Status}
 * @example <caption>Hook body example</caption>
 * {
 *      project: '72',
 *      invoice: '100549706',
 *      status: 'Paid',
 *      amount: '5.00',
 *      net_amount: '5.00',
 *      // yup, they have the misspell in word "received"
 *      recieved_amount: '4.60',
 *      user: '13',
 *      type: 'yandex',
 *      wallet: '410012389058372',
 *      comment: '',
 *      currency_project: 'RUB',
 *      currency_user: 'RUB',
 *      time: '1576885403',
 *      signature: 'very-long-base64-string, nearly 500 chars'
 * }
 *
 *
 * {
     amount:  "10"
     comment:  ""
     currency_project:  "RUB"
     currency_user:  "RUB"
     invoice:  "117813963"
     net_amount:  "10"
     project:  "147"
     recieved_amount:  "9.5"
     signature:  'very-long-base64-string, nearly 500 chars',
     status:  "Paid"
     time:  "1588152109"
     type:  "alfabank"
     user:  "54"
     wallet:  ""
    }
 */



const handleDepositHook = async (req, res) => {
    try {

        console.log('gamemoney deposit hook body:', req.body)

        /** body validation */
        if (!req.body.project ||
            !req.body.invoice ||
            !req.body.status ||
            !req.body.amount ||
            !req.body.net_amount ||
            /** that's the spell mistake in the  property name */
            !req.body.recieved_amount ||
            !req.body.user ||
            !req.body.type ||
            !req.body.currency_project ||
            !req.body.currency_user ||
            !req.body.time ||
            !req.body.signature ) {
            return res.status(424).json({status: codes.WRONG_ARGS})
        }

        /**
         * @type {{status: Status, deposit_id: String}}
         */
        const handling_result = await GameMoneyLib.handleDepositHook({
            ...req.body,
            /** @ignore
             * @desc correcting the spell mistake of gamemoney platform authors */
            received_amount: req.body.recieved_amount
        })



        if (codes.statusEqual(handling_result.status, codes.OK)) {



            const {user_id} = handling_result

            if (user_id) {

                const balance = await DB.query(`SELECT calcBalance($1) as balance`, [
                    user_id
                ]).then(r => r.rows[0] ? r.rows[0].balance : null)

                if (balance !== null) {
                    nats.publish('PERSONAL.ACCOUNT.BALANCE_UPDATE', {user_id: String(user_id), balance})
                }



            }


            if (config.GAMEMONEY_DEPOSIT_STATUSES_DICTIONARY[req.body.status] === 'COMPLETED') {
                handleNewCompletedDeposit({...handling_result, deposit_amount: handling_result.net_amount_usd})
            }


            /**
             * @ignore
             * @desc The hooks not responded with {success: 'true'} will be retried
             * @see {@link https://cp.gamemoney.com/apidoc#anchor3}
             */
            return res.status(200).json({ success: 'true', status: codes.OK })
        } else {
            return res.status(500).json({ success: 'false', status: codes.FAIL })
        }



    } catch (e) {
        error(`${req.originalUrl} err:`, e)
        /** we return 500 http-server status instead of 200 cause we want gamemoney to repeat the hook */
        return res.status(500).json({success: false, status: codes.FAIL})
    }
}

/**
 * It handles the withdraws status updating hooks from gamemoney backend
 * @chain {@link module:http-server/api/payments/gamemoney~gamemoneyHookSignatureVerificationMiddleware}
 * @chain This endpoint
 * @function handleDepositHook
 * @path {POST} /api/payments/gamemoney/deposit-hook
 * @code {200} if hook handled correctly
 * @code {500} if internal error - hook should be repeated in future
 */
const handleWithdrawHook = async (req, res) => {
    try {

        info('withdraw hook body:', req.body)

        return res.status(424).json({status: codes.TEMPORARY_UNAVAILABLE})

    } catch (e) {
        error(e)
        /** we return 500 http-server status instead of 200 cause we want gamemoney to repeat the hook */
        return res.status(500).json({status: codes.FAIL})
    }
}


/**
 * It responds with the our project id and generated unique signature if the provided deposit amount is correct.
 * You need this data to add them to form we send to  the payments provider making the redirect to pay.
 * Also it validates provided amount to be in between {@link module:http-server/config.MIN_DEPOSIT_AMOUNT }
 * and {@link module:http-server/config.MAX_DEPOSIT_AMOUNT }
 * @function generateDepositRedirectSignature
 * @code {200} always
 * @path {POST} /api/payments/gamemoney/generate-deposit-redirect-signature
 * @chain {@link module:http-server/middleware/access~addAuthorizationDataToRequest}
 * @chain {@link module:http-server/middleware/access~accessUser}
 * @chain this endpoint
 * @body {Number} amount
 * @body {String} [success_url] - url to redirect if deposit was completed successful
 * @body {String} [fail_url] - url to redirect if deposit not completed
 * @response {Status} project_id if status is OK
 * @response {Status} signature if status is OK
 * @response {Status} status WRONG_ARGS if amount not provided or isn't able to cast to the numeric value
 * @response {Status} status WRONG_ARGS if signature generated properly
 * @response {Status} status payments.INCORRECT_DEPOSIT_AMOUNT if amount validation failed
 * @response {Number} MIN_DEPOSIT_AMOUNT if amount validation failed
 * @response {Number} MAX_DEPOSIT_AMOUNT if amount validation failed
 * @response {Status} status FAIL if internal error
 */
const generateFiatDepositRedirectSignature = (req, res) => {
    try {

        if (!req.body.amount) {
            return res.json({status: codes.WRONG_ARGS})
        }

        /**
         * The RUR amount which user have to deposit
         * @ignore
         * @type {number}
         */
        const amount = parseInt(req.body.amount)

        if (isNaN(amount)) {
            return res.json({status: codes.WRONG_ARGS})
        }


        /**
         * Amount validation using min/max deposit
         * amount values from config
         */
        if (amount < config.MIN_DEPOSIT_AMOUNT ||
            amount > config.MAX_DEPOSIT_AMOUNT) {
            return res.json({
                status: codes.payments.INCORRECT_DEPOSIT_AMOUNT,
                MIN_DEPOSIT_AMOUNT: config.MIN_DEPOSIT_AMOUNT,
                MAX_DEPOSIT_AMOUNT: config.MAX_DEPOSIT_AMOUNT
            })
        }

        const deposit_data = {
            amount,
            project: Number(config.GAMEMONEY_PROJECT_ID),
            user: req.account_data.user_id
        }

        if (typeof req.body.success_url === 'string') {
            deposit_data.success_url = req.body.success_url
        }

        if (typeof req.body.fail_url === 'string') {
            deposit_data.fail_url = req.body.fail_url
        }

        /**
         * Creating a signature using gamemoney SDK
         * @see {@link https://cp.gamemoney.com/apidoc#anchor9}
         * @type {string}
         * @ignore
         */
        const signature = GameMoneyLib.generateHmacSignature(deposit_data)

        return res.json({ status: codes.OK,
            signature,
            project_id: config.GAMEMONEY_PROJECT_ID
        })

    } catch (e) {
        error(e)
        return res.json({status: codes.FAIL})
    }
}







router.post('/generate-fiat-deposit-redirect-signature', accessUser, generateFiatDepositRedirectSignature)
