router.use(function (req, res, next) {

    /**
     * On 'development' stage we return the wildcard (*), on 'staging' / 'test' / 'production' we return exact the req origin as an Access-Control-Allow-Origin
     * @ignore
     */

    const referer = (() => {

        if (process.env.NODE_ENV === 'development') {

            if (req.headers.origin) {
                return req.headers.origin
            } else {
                return '*'
            }

        } else {


            if (req.DOMAIN) {
                return req.DOMAIN
            } else if (req.headers.origin) {
                return req.headers.origin
            } else {
                return '*'
            }

        }

    })()

    res.setHeader('Vary', 'Origin')

    res.setHeader('Access-Control-Allow-Origin', referer)

    res.setHeader('Access-Control-Allow-Credentials', true)

    res.setHeader('Access-Control-Allow-Methods', 'GET, POST, OPTIONS, PUT, PATCH, DELETE')

    res.setHeader('Access-Control-Allow-Headers', 'Content-Type, Access-Control-Allow-Headers, Authorization, X-Requested-With')

    next()
})

/**
 * @function startServe
 * @summary it starts the server on the port from env
 */
const startServe = () => {

    if (!process.env.PORT) {
        throw new Error('port to listen is not specified. Plz set .PORT in env')
    }

    /**
     * @var process.env.PORT {Number|String}
     * is a port which we should listen
     */

    const server = http.createServer(app)

    /**
     * It is a temporary var to store the http-server object
     * before its method .close() to the 'app' object in code below */
    server.listen(process.env.PORT, (err) => {
        if (err) throw err
        info(`http-server started on ${process.env.PORT}`)
        HealthCheck.updateState('UP')
    })

    /** I dont know why I can not just assign server.close to app.close
     * but it doesnt works
     * @returns {*}
     */
    app.close = () => server.close()
}

/**
 * it shuts the server down
 * ACHTUNG! If the event loop become empty, the process will end with exit code 0
 * @method shutDown
 */
const shutDown = () => {
    if (!app.close) throw new Error('You tried to stop the server which isn`t up')
    app.close()
    if (HealthCheck.getState() !== 'SHUTTING_DOWN') HealthCheck.updateState('SHUTTING_DOWN')
}


/**
 * ACHTUNG! The http-server server won't be started automatically
 * if it is in the test env (env.NODE_ENV === 'test')
 * It was done for testing purposes (to start and stop the server for each test case)
 */
if (process.env.NODE_ENV !== 'test') startServe()


/**
