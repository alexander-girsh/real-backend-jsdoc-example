/**
 * @module Inventory
 */

/**
 * @desc ENUM, existed steam app names, for e.g "DOTA2"
 * @typedef {String} SteamAppName
 * @property {String} DOTA2
 * @property {String} CS_GO
 */

/**
 * @desc ENUM, known values for InventoryItem.status, for e.g "STORED"
 * @typedef {String} InventoryItemStatus
 * @property {String} STORED - an item which may be placed as a bet, sold or withdrawn
 * @property {String} WITHDRAWAL_REQUEST_CREATED - item with a created withdraw request. request may be rejected from admin or removed by user. item will become "STORED".
 * @property {String} WITHDRAWAL_REQUEST_APPROVED - an item whose withdrawal request has been approved by the admin. the user can no longer cancel the request.
 * @property {String} WITHDRAW_PROCESSING - the item whose withdrawal request is currently being processed on the backend
 * @property {String} AWAITING_WITHDRAW_CONFIRMATION - the state when the backend has already processed the withdrawal, but the user must accept the exchange on the steam site
 * @property {String} WITHDRAWN - item was withdrawn completely
 * @property {String} SOLD - item was placed as a bet in the game or sold using the items-selling api
 */

/**
 * @typedef InventoryItem
 * @property {String} market_hash_name - an unique identifier of the steam item
 * @property {SteamAppName} steam_app_name
 * @property {Number} price - in usd
 * @property {BigInt} inventory_item_id - an unique id of the InventoryItem
 * @property {InventoryItemStatus} status - current status, for e.g "STORED" or "WITHDRAWN"
 */

/**
 * @module Inventory
