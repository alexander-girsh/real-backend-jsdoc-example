http-server/plugins/affiliates/revshare.js

const {error} = require('../../../logger')
const {codes} = require('status-codes')
const {execRepeatableTransaction} = require('../../../database/connection.js')
const {config} = require('../../config.js')
const {createDefaultTariff} = require('./tariffs.js')


/*
Алгоритм выполнения начислений по ревшаре

1.

    Если аккаунтинг инаута выполняется "по партнёрскому аккаунту",
    то вычисляются тотал инаут и сумма выполненных начислений по тотал инауту.
    если тотал инаут отрицательный - начисления не происходит,
    если сумма выполненных начислений по тотал инауту превышает текущий тотал инаут,
    то начисления тоже не происходит.

    Если аккаунтинг инаута выполняется "по пользователю", то шаги выше пропускаются.

    Вычисляется инаут пользователя и сумма выполненных начислений по инауту пользователя,
    если инаут пользователя отрицательный или сумма выполненных начислений по инауту пользователя
    превышает текущий инаут пользователя, то начисление не происходит.

    Если сумма на балансе пользователя превышает минимальную стоимость предмета на внутреннем маркете или в инвентаре пользователя есть предметы,
    то начисляется % от инаута, подлежащий начислению. Процент берётся из свойства .revshare_pace_of_accruing текущего тарифа.

    Если сумма на балансе меньше, чем минимальная стоимость предмета и в инвентаре пользователя нет предметов,
    начисляется вся сумма, подлежащая начислению по текущему пользователю.

    ---

    Процесс вычисления выполненных начислений по текущему инауту (вычисления доли оплаченного инаута в общем инауте)
    или "сколько от инаута мы оплатили партнёру, а сколько ещё должны оплатить".

    Берутся все подходящие начисления (в зависимости от того, считаем по пользователю или по аккаунту).

    Начисления содержат ссылку на тариф, который был актуален на момент их выполнения (.affiliate_tariff_id).

    Тариф содержит процент (.revshare_percentage), исходя из которого в момент начисления рассчитывался инаут.

    Нам остаётся вычислить 100% в $, зная .revshare_percentage и .accrued_amount_usd.

    Сумма этого ряда чисел и даст долю "оплаченного" инаута.

 */


/**
 *
 * @param {String} user_id
 * @returns {Promise<{status: Status}|{status: Status, desc: string}|{status: Status, accrued_usd_amount: Number}>}
 */
const performRevshareAccrual = async ({user_id}) => {
    try {


        if (!user_id) {
            return {status: codes.WRONG_ARGS, desc: `.user_id not provided or invalid: ${user_id}`}
        }

        return await execRepeatableTransaction(async (client) => {

            const referer_affiliate = await client.query(`SELECT
                    uab.*,
                    (SELECT ( SELECT row_to_json(matching_tariff) FROM (     SELECT at.*, at.affiliate_id::BIGINT, (at.valid_before::TIMESTAMPTZ >= now()::TIMESTAMPTZ) AS is_still_valid
                                                                             FROM affiliate_tariffs at
                                                                             WHERE at.affiliate_id = uab.affiliate_id
                                                                               AND at.tariff_kind = uab.tariff_kind
                                                                               AND at.tariff_kind = 'REVSHARE'
                                                                             ORDER BY is_still_valid DESC
                                                                             LIMIT 1
                                                                                ) matching_tariff ) 
                    ) AS matching_tariff
                    FROM user_affiliate_bindings uab WHERE user_id = $1`, [
                user_id
            ])
                .then(r => r.rows[0])

            if (!referer_affiliate) {
                return {status: codes.NOT_FOUND, desc: `user #${user_id} has no affiliate binding`}
            }

            if (!referer_affiliate.matching_tariff.is_still_valid) {


                /** creating new tariff if there is no one with .valid_before */
                if (config.DEFAULT_AFFILIATE_TARIFF_KIND === 'REVSHARE') {

                    const {status, tariff} =  await createDefaultTariff({affiliate_id: referer_affiliate.affiliate_id})

                    if (!codes.statusEqual(status, codes.OK)) {
                        return {status, desc: 'Unable to create new revshare tariff and the existing ones are expired. See .status field.'}
                    }


                }

                return {status: codes.INCORRECT_STATE, desc: `affiliate #${referer_affiliate.affiliate_id} has no valid matching tariff (.tariff_kind = REVSHARE)`}
            }

            /**
             * If .revshare_inout_accounting is 'BY_AFFILIATE_ACCOUNT'
             * and already accrued profit by revshare is greater
             * than sum of funds which should be accrued totally,
             * we should break the accrual.
             */
            if (referer_affiliate.matching_tariff.revshare_inout_accounting ===  'BY_AFFILIATE_ACCOUNT') {
                const {not_payed_off_affiliate_revshare_inout} = await client.query(`SELECT 
                    calcNotPayedOffAffiliateRevshareInout($1, NULL) AS not_payed_off_affiliate_revshare_inout`,
                [
                    referer_affiliate.affiliate_id
                ])
                    .then(r => r.rows[0])

                if (not_payed_off_affiliate_revshare_inout <= 0) {
                    return {status: codes.INCORRECT_STATE, desc: `Not payed off total affiliate account inout is less then $0 ($${not_payed_off_affiliate_revshare_inout})`}
                }
            }


            const {not_payed_off_inout_by_user_account} = await client.query(`
                SELECT calcNotPayedOffAffiliateRevshareInout($1, $2) AS not_payed_off_inout_by_user_account
                        
            `, [
                referer_affiliate.affiliate_id,
                user_id
            ])
                .then(r => r.rows[0])

            if (not_payed_off_inout_by_user_account <= 0) {
                return {status: codes.INCORRECT_STATE, desc: `Not payed off user account inout is less then $0 ($${not_payed_off_inout_by_user_account})`}
            }


            const usd_amount_to_be_accrued = await ( async () => {


                const {
                    user_balance,
                    available_inventory_items_count,
                    withdrawn_inventory_items_count
                } = await client.query(`
                    SELECT calcBalance($1) as user_balance,
                           count(*)::INT AS available_inventory_items_count,
                           (SELECT 
                                   count(*)::INT 
                            FROM inventory_items ii2 
                                WHERE ii2.user_id = $1 
                            AND status = 'WITHDRAWN'
                            ) AS withdrawn_inventory_items_count
                    FROM inventory_items 
                    WHERE user_id = $1 
                      AND status NOT IN ('SOLD', 'WITHDRAWN')
                      AND price >= $2::NUMERIC(16,2)`, [
                    user_id,
                    config.MIN_BET_AMOUNT_USD.toFixed(2)
                ])
                    .then(r => r.rows[0])


                if (user_balance <= config.MIN_BET_AMOUNT_USD &&
                    available_inventory_items_count === 0 &&
                    withdrawn_inventory_items_count === 0) {

                    /** user can not do more bets until the next deposits */
                    return not_payed_off_inout_by_user_account

                } else {

                    /** user can do additional bets. The amount to accrue calculates using speed value from current affiliate tariff
                     *  Measuring value is the percentage of not payed off user inout
                     * */
                    return Math.max(0.01, parseFloat(((not_payed_off_inout_by_user_account / 100) * referer_affiliate.matching_tariff.revshare_pace_of_accruing).toFixed(2)))
                }

            })()


            if (!usd_amount_to_be_accrued) {
                return {status: codes.INCORRECT_STATE, desc: `Usd amount to be accrued is $${usd_amount_to_be_accrued}`}
            }

            await client.query(`
                INSERT INTO affiliate_accruals (affiliate_id, accrued_amount_usd, affiliate_tariff_id, user_id)
                VALUES ($1, $2, $3, $4)`, [
                referer_affiliate.affiliate_id,
                usd_amount_to_be_accrued,
                referer_affiliate.matching_tariff.affiliate_tariff_id,
                user_id
            ])

            return {status: codes.OK, accrued_usd_amount: usd_amount_to_be_accrued}

        })


    } catch (e) {
        error('performAffiliateAccrual() err:', e)
        return {status: codes.FAIL}
    }
}


module.exports = {performRevshareAccrual}
