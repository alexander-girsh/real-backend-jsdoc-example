/**
 * @desc enables SameSite = 'Strict', secure and httpOnly flags for cookie including the auth cookie
 * @member COOKIE_SECURE_MODE
 * @type {Boolean}
 */
config.COOKIE_SECURE_MODE = (() => {

    if (!process.env.COOKIE_SECURE_MODE) {
        return undefined
    }

    if (process.env.COOKIE_SECURE_MODE.trim() === 'true') {
        return true
    }

    if (process.env.COOKIE_SECURE_MODE.trim() === 'false') {
        return false
    }

    return undefined

})()

/**
 * @member AUTH_COOKIE_OPTIONS
 * @see {@link module:http-server/middleware/access~setAuthCookie}
 * @type {{maxAge: number, signed: boolean, httpOnly: boolean}}
 * @type Object
 * @property maxAge {Number} milliseconds before cookie expiration, for e.g 3600 * 24 * 14000 means two weeks
 * @property signed {Boolean} directive to sigh cookie using {@link module:http-server/config~COOKIES_SECRET} key
 * @property httpOnly {Boolean} directive to disallow client-side JS to CRUD this cookie
 */
config.AUTH_COOKIE_OPTIONS = config.COOKIE_SECURE_MODE ?
    // secure mode
    {
        maxAge: 3600 * 24 * 14 * 1000,
        httpOnly: true,
        secure: true,
        signed: true,
        SameSite: 'Strict',
        sameSite: 'Strict'
    }
    :
    // not secure mode
    {
        maxAge: 3600 * 24 * 14 * 1000,
        httpOnly: false,
        secure: false,
        signed: true,
        SameSite: 'Lax',
        sameSite: 'Lax'
    }
