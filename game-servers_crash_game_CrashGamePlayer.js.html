const {nats} = require( 'nats-connection/index.cjs')
const {error, info} = require( '../../../logger/index.js')
const {GameWin} = require( './GameWin.js')
const {codes} = require( 'status-codes')
const {GameBet} = require( './GameBet.js')
const {InventoryItem} = require( './InventoryItem.js')
const util = require( 'util')
const {CrashGameMultiplier} = require( './CrashGameMultiplier.js')
const {CrashGameDBTransaction} = require('./CrashGameDBTransaction.js')

class User {


    /**
     * @type { String }
     */
    #user_id

    /**
     * @type { String }
     */
    #username

    /**
     * @type { String | Null }
     */
    #avatar_link

    constructor ({user_id, username, avatar_link = null}) {

        if (!user_id || !username) {
            throw new Error(`Not all required params (user_id, username) provided: ${util.inspect({user_id, username})}`)
        }

        this.#user_id = user_id
        this.#username = username
        this.#avatar_link = avatar_link || null
    }


    get userId () {
        return this.#user_id
    }

    get username () {
        return this.#username
    }

    get avatarLink () {
        return this.#avatar_link
    }


}



class CrashGamePlayer extends User {

    /**
     * @type { GameBet }
     */
    #bet


    /**
     * @type { GameWin }
     */
    #win


    /**
     * @type {CrashGameMultiplier}
     */
    #auto_out_at_multiplier

    /**
     * @type {Boolean}
     */
    #out_directive_added = false

    /**
     * @type {boolean}
     */
    #out_processing_started = false

    /**
     * @type {CrashGameMultiplier}
     */
    #out_at_multiplier

    /**
     * @type {CrashGameDBTransaction.client}
     */
    #client




    constructor ({bet, auto_out_at_multiplier, ...other_params}, {client}) {

        super({...other_params})


        if (!bet) {
            throw new Error(`bet argument is required. Passed: ${util.inspect({bet, auto_out_at_multiplier, ...other_params})}`)
        }

        this.#bet = bet
        this.#auto_out_at_multiplier = new CrashGameMultiplier(auto_out_at_multiplier)

        this.#client = client

    }

    /**
     * @param game_id
     * @param {User.user_id} user_id
     * @param {InventoryItem.inventory_item_id[]} inventory_item_ids
     * @param {Number} auto_out_at_multiplier
     * @param {CrashGameDBTransaction} db_transaction
     * @return {Promise<{status: *}>}
     */
    static async create  ({user_id, game_id, inventory_item_ids, auto_out_at_multiplier }, {db_transaction}) {

        try {

            if (!game_id || !user_id || !inventory_item_ids || inventory_item_ids.some(i => !i) || !auto_out_at_multiplier || parseFloat(auto_out_at_multiplier.toFixed(2)) <= 1.00 ) {
                return {status: codes.WRONG_ARGS}
            }

            if (!db_transaction || !(db_transaction instanceof CrashGameDBTransaction)) {
                return {status: codes.WRONG_ARGS, desc: {db_transaction}}
            }

            const random_client = db_transaction.randomClient({user_id})



            const a = Date.now()

            const {avatar_link, username, inventory_item_objects} = await random_client.query(`
                SELECT avatar_link, 
                       username,
                       ARRAY( SELECT row_to_json(bet_inventory_items) FROM  (
                           SELECT si.*, ii.* FROM inventory_items ii
                              LEFT JOIN steam_items si
                                 ON (ii.market_hash_name = si.market_hash_name)
                           WHERE inventory_item_id = ANY($2::BIGINT[]) 
                                  AND ii.status = 'STORED'
                                  AND ii.user_id = $1) bet_inventory_items
                       ) as inventory_item_objects
                FROM users WHERE user_id = $1
            `, [
                user_id,
                inventory_item_ids
            ]).then(r => r.rows[0])

            console.log(`a: ${Date.now() - a}ms`, random_client.queryQueue.length)

            if (!username) {
                return {status: codes.NOT_FOUND}
            }

            const inventory_items = inventory_item_objects.map(i_i_o => InventoryItem.fromObject(i_i_o))


            if (inventory_items.length !== inventory_item_ids.length) {
                return {status: codes.INCORRECT_STATE}
            }

            let known_steam_app_name = null
            for (const {steam_app_name} of inventory_items) {
                if (known_steam_app_name === null) {
                    known_steam_app_name = steam_app_name
                } else if (steam_app_name !== known_steam_app_name) {
                    /** inventory items have various values of .steam_app_name field */
                    return {status: codes.IMPOSSIBLE}
                }
            }

            const bet_creation_result = await GameBet.create({inventory_items, user_id, game_id, auto_out_at_multiplier}, {client: random_client})


            if (!codes.statusEqual(bet_creation_result.status, codes.OK)) {
                return {status: bet_creation_result.status}
            }

            const {bet} = bet_creation_result

            return {
                status: codes.OK,
                player: new CrashGamePlayer({bet, auto_out_at_multiplier, user_id, avatar_link, username}, {client: random_client})
            }


        } catch (e) {
            error('games/crash/join() err:', e)
            return {status: codes.FAIL}
        }
    }




    /**
     * @returns {CrashGameMultiplier}
     */
    get autoOutAtMultiplier () {
        return this.#auto_out_at_multiplier
    }

    /**
     * @returns {CrashGameMultiplier}
     */
    get outAtMultiplier () {
        return this.#out_at_multiplier
    }


    get publicInfo () {
        return {
            game_id: this.bet.gameId,
            user_id: this.userId,
            username: this.username,
            avatar_link: this.avatarLink,
            bet: this.bet.publicInfo,
            win: this.win ? this.win.publicInfo : null,
            out_at_multiplier: this.outAtMultiplier ? this.outAtMultiplier.to2NumbersPrecisionFloat : null
        }
    }


    /**
     * @param {CrashGameMultiplier} out_multiplier
     */
    async makeOut (out_multiplier,  ) {

        try {

            if (this.#out_processing_started) {
                return {status: codes.ALREADY_DONE}
            }

            this.#out_processing_started = true

            this.#out_at_multiplier = out_multiplier

            const {status, win} = await GameWin.create(this, {client: this.#client})

            if (!codes.statusEqual(status, codes.OK)) {
                error(`GameWin.create() returned ${status}`)
                return {status: codes.FAIL}
            }

            this.#win = win

            const game_results = {
                game_id: this.bet.gameId
            }

            if (this.win.wonFiatAmount) {
                game_results.updated_balance =
                    await this.#client.query(`SELECT calcbalance(user_id) as updated_balance FROM users WHERE user_id = $1`, [this.userId])
                        .then(r => r.rows[0].updated_balance)
            }

            if (this.win.wonInventoryItem) {
                game_results.won_inventory_item = this.win.wonInventoryItem.publicInfo
            }

            nats.publish('PERSONAL.GAMES.CRASH.GAME_RESULTS', {user_id: this.userId, ...game_results})

            nats.requestOne({
                subj: 'AFFILIATES.REVSHARE_ACCRUAL_TRIGGER',
                msg: {user_id: this.userId},
                retries: 10,
                opts: {queue: 'AFFILIATES.REVSHARE_ACCRUAL_TRIGGER'}
            })

            return {status: codes.OK}

        } catch (e) {
            error('games/crash/player/makeOut() err:', e)
            return {status: codes.FAIL}
        }

    }

    addOutDirective () {
        this.#out_directive_added = true
    }

    get win () {
        return this.#win
    }

    get bet () {
        return this.#bet
    }


    get outDirectiveAdded () {
        return this.#out_directive_added
    }


    get outProcessingStarted () {
        return this.#out_processing_started
    }

}
