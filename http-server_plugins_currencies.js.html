http-server/plugins/currencies.js

const {DB} = require( '../../database/connection.js')
const axios = require( 'axios')
const {codes} = require( 'status-codes')
const Iconv = require( 'iconv')
const XML2Js = require( 'xml2js')
const {config} = require( '../config.js')
const {asyncInterval} = require( '../../lib.js')
const {error, info} = require( '../../logger/index.js')

const USD_EXCHANGE_RATE = {
    market_rur_rate: null,
    /**
     *
     * @returns {Promise<Number>}
     */
    async getTransformedRate () {
        if (!this.market_rur_rate) {
            return new Promise((resolve, reject) => {
                DB.query(`SELECT rur_rate FROM currencies WHERE currency_name = 'USD'`).then( r => {

                    if (!r.rows[0] || !r.rows[0].rur_rate) {
                        reject(new Error(`Unable to get the exchange rate from DB`))
                    }

                    this.market_rur_rate = r.rows[0].rur_rate
                    resolve(this.transformedUSDExchangeRate())
                })
            })
        }

        return this.transformedUSDExchangeRate()
    },
    async getMarketRate () {
        if (!this.market_rur_rate) {
            await this.getTransformedRate()
        }
        return this.market_rur_rate
    },
    /**
     * @returns Number
     * */
    transformedUSDExchangeRate () {
        return parseFloat((this.market_rur_rate * config.USD_EXCHANGE_RATE_TRANSFORM_COEFFICIENT).toFixed(2))
    },
    async exchangeRURtoUSD (rur_amount) {
        try {
            if (!rur_amount || typeof rur_amount !== 'number') {
                return {status: codes.WRONG_ARGS}
            }
            const rate = await this.getTransformedRate()

            const usd_amount = parseFloat( (rur_amount / rate).toFixed(2) )

            return {status: codes.OK, usd_amount}

        } catch (e) {
            error('USD_EXCHANGE_RATE.exchangeRURtoUSD err:', e)
            return {status: codes.FAIL}
        }
    },
    async fetchNewMarketRate () {
        try {

            /** cause the cbr.ru API responds in windows-1251 encoding, we should decode it manually from a binary blob */
            const {headers, data} = await axios.get('https://www.cbr.ru/scripts/XML_daily.asp', {
                responseType: 'arraybuffer'
            })

            /** usually a 'windows-1251' */
            const charset = headers['content-type'].split('charset=')[1].split(';')[0]


            if (!charset) {
                throw new Error(`Unable to parse the www.cbr.ru response. Unable to get charset. Response headers: ${JSON.stringify(headers)}`)
            }

            const converter = Iconv.Iconv('windows-1251', 'utf8')


            /** so we have the XML =) */
            const data_in_utf8 = converter.convert(data).toString()

            const response_object = await XML2Js.parseStringPromise(data_in_utf8)

            const USD_currency_data_object = response_object['ValCurs']['Valute'].filter(currency => currency.CharCode[0] === 'USD')[0]

            if (!USD_currency_data_object || !USD_currency_data_object.Value || !Array.isArray(USD_currency_data_object.Value) ) {
                throw new Error(`Unable to parse the www.cbr.ru response. Unable to get the valid USD currency data object. Got ${JSON.stringify(USD_currency_data_object)}`)
            }

            const USD_exchange_rate_string = USD_currency_data_object['Value'][0]

            if (!USD_exchange_rate_string) {
                throw new Error(`Unable to parse the www.cbr.ru response. Unable to get exchange rate string. Got currency data object: ${JSON.stringify(USD_currency_data_object)}`)
            }

            const RUR_rate = parseFloat(USD_exchange_rate_string.replace(',', '.')).toFixed(2)

            if (!RUR_rate) {
                throw new Error(`Unable to parse the www.cbr.ru response. Unable to format the resulting rate. Got resulting rate: ${JSON.stringify({USD_exchange_rate_string, RUR_rate})}`)
            }

            await DB.query(`INSERT INTO currencies (currency_name, rur_rate, last_updated_at) VALUES ('USD', $1, now())
            ON CONFLICT (currency_name) DO UPDATE SET rur_rate = $1, last_updated_at = now() RETURNING *`, [
                RUR_rate
            ])

            this.market_rur_rate = RUR_rate

            info(`plugins/currencies/USD_EXCHANGE_RATE new USD/RUR exchange rate fetched`)

            return {status: codes.OK}


        } catch (e) {
            console.error(`plugins/currencies/USD_EXCHANGE_RATE.fetchNewUSDExchangeRate() err:`, e)
            return {status: codes.FAIL}
        }
    },
    startRateRenewLoop () {
        asyncInterval(async () => {
            await this.fetchNewMarketRate()
        }, 3 * 60 * 60 * 1000 /** 3 hours */ )
    },
    init () {
        /** init value */
        DB.query(`SELECT rur_rate FROM currencies WHERE currency_name = 'USD'`).then( r => {

            if (r.rows[0] && r.rows[0].rur_rate) {
                this.market_rur_rate = r.rows[0].rur_rate
            }
        })

        this.startRateRenewLoop()
    }
}
