const {DB, execRepeatableTransaction} = require('../../../database/connection.js')
const {error, info} = require('../../../logger')
const {codes} = require('status-codes')
const {config} = require('../../config.js')
const {nats} = require('nats-connection/index.cjs')
const {getValidMatchingTariff} = require('./tariffs.js')
const {performRevshareAccrual} = require('./revshare.js')

const AFFILIATES_LIST = {}

const addAffiliatesToList = async ({affiliate_id = null} = {}) => {
    await DB.query(`SELECT *, 
                                  ARRAY( 
                                      SELECT 
                                             row_to_json(tariffs) 
                                      FROM (
                                          SELECT * 
                                          FROM affiliate_tariffs 
                                          WHERE 
                                                valid_before > now() 
                                            AND affiliate_id = a.affiliate_id 
                                            ) tariffs 
                                      ) as tariffs 
                            FROM affiliates a
                            WHERE $1::BIGINT IS NULL OR affiliate_id = $1::BIGINT`, [
        affiliate_id
    ])
        .then(r => r.rows)
        .then(affiliates => {
            affiliates.forEach(affiliate => {
                AFFILIATES_LIST[affiliate.affiliate_id] = affiliate
            })
        })
}

nats.subscribe('AFFILIATES.NEW_AFFILIATE_CREATED', async ({affiliate_id, force_renew = true}) => {

    if (!force_renew && affiliateExists({affiliate_id})) {
        return
    }

    await addAffiliatesToList({affiliate_id})

/**
 * @param {String} affiliate_id
 * @returns {Boolean}
 */
const affiliateExists = ({affiliate_id}) => {

    return Boolean(AFFILIATES_LIST[affiliate_id])
}

/**
 * @param {String} affiliate_id
 * @returns {Promise<{status: Status}|{status: Status, affiliate: Object}>}
 */
const getAffiliate = ({affiliate_id}) => {

    if (!affiliate_id) {
        return {status: codes.WRONG_ARGS}
    }


    if (affiliateExists({affiliate_id})) {
        return {
            status: codes.OK,
            affiliate: AFFILIATES_LIST[affiliate_id]
        }
    }

    return {status: codes.NOT_FOUND}
}

const createNewAffiliate = async ({user_id}) => {
    try {

        if (!user_id) {
            return {status: codes.WRONG_ARGS}
        }


        const {status, affiliate} =  await execRepeatableTransaction(async (client) => {

            const new_affiliate = await client.query(`
                INSERT INTO affiliates (affiliate_id) 
                (SELECT user_id FROM users WHERE user_id = $1)
                RETURNING *`, [
                user_id
            ])
                .then(r => r.rows[0])


            if (!new_affiliate) {
                error(`Unable to create new affiliate using provided .user_id "${user_id}"`)
                return {status: codes.FAIL}
            }


            const default_tariff_for_new_affiliate = await client.query(`INSERT INTO affiliate_tariffs (
                                                                           affiliate_id,
                                                                           valid_before,
                                                                           tariff_kind,
                                                                           revshare_percentage,
                                                                           revshare_inout_accounting,
                                                                           revshare_pace_of_accruing) 
                                                             VALUES (
                                                                           $1,
                                                                            /** default tariff will be valid until the first second of next month */
                                                                            (SELECT date_trunc('month', now() + interval '1 month')::DATE),
                                                                           $2,
                                                                           $3, 
                                                                           $4, 
                                                                           $5
                                                                    ) RETURNING *`, [
                new_affiliate.affiliate_id,
                config.DEFAULT_AFFILIATE_TARIFF_KIND,
                config.DEFAULT_REVSHARE_PERCENTAGE,
                config.DEFAULT_REVSHARE_INOUT_ACCOUNTING,
                config.DEFAULT_REVSHARE_PACE_OF_ACCRUING
            ])
                .then(r => r.rows[0])

            if (!default_tariff_for_new_affiliate) {
                error(`Unable to create new tariff for new affiliate (.user_id is "${user_id}")`)
                return {status: codes.FAIL}
            }


            new_affiliate.tariffs = [default_tariff_for_new_affiliate]


            return {
                status: codes.OK,
                affiliate: new_affiliate
            }

        })

        if (!codes.statusEqual(status, codes.OK)) {
            return {status}
        }

        AFFILIATES_LIST[affiliate.affiliate_id] = affiliate

        nats.publish('AFFILIATES.NEW_AFFILIATE_CREATED', {
            affiliate_id: affiliate.affiliate_id,
            force_renew: false
        })

        return {status, affiliate}

    } catch (e) {
        error(`createNewAffiliate() err:`, e)
        return {status: codes.FAIL}
    }
}


const bindUserToAffiliate = async ({user_id, affiliate_id, affiliate_tariff_id = null}) => {
    try {

        if (!user_id || !affiliate_id) {
            return {status: codes.WRONG_ARGS}
        }

        return await execRepeatableTransaction(async (client) => {


            const {deposits_count, received_promocode_bonuses_count, bets_count} = await client.query(`
                        SELECT 
                        (SELECT COALESCE(count(*)::INT, 0) FROM deposits WHERE user_id = $1 AND status = 'COMPLETED') AS deposits_count,
                        (SELECT COALESCE(count(*)::INT, 0) FROM promocodes_activations WHERE user_id = $1 AND status = 'COMPLETED') AS received_promocode_bonuses_count,
                        (SELECT COALESCE(count(*)::INT, 0) FROM bets WHERE user_id = $1) AS bets_count
                    `, [
                user_id
            ]).then(r => r.rows[0])

            if (deposits_count > 0 ||
                received_promocode_bonuses_count > 0 ||
                bets_count > 0) {
                info(`User #${user_id} was not bound to affiliate #${affiliate_id} because he already has ${deposits_count} deposits, ${received_promocode_bonuses_count} promocode bonuses and ${bets_count} bets`)
                return {status: codes.INCORRECT_STATE}
            }


            let tariff = null

            const provided_tariff = await client.query(`
                SELECT *,
                       valid_before <= now()::TIMESTAMPTZ AS is_expired
                FROM affiliate_tariffs 
                WHERE affiliate_id  = $1 
                  AND affiliate_tariff_id = $2`, [
                affiliate_id,
                affiliate_tariff_id
            ])
                .then(r => r.rows[0])

            /** provided tariff is expired or does not exist */
            if (!provided_tariff || provided_tariff.is_expired) {

                const {affiliate} = await getAffiliate({affiliate_id})

                if (!affiliate) {
                    return {status: codes.NOT_FOUND}
                }

                tariff = (await getValidMatchingTariff({
                    affiliate,
                    tariff_kind: provided_tariff  ? provided_tariff.tariff_kind : config.DEFAULT_AFFILIATE_TARIFF_KIND
                })).tariff

            } else {

                tariff = provided_tariff

            }

            if (!tariff) {
                return {status: codes.IMPOSSIBLE}
            }

            const result = await client.query(`INSERT INTO user_affiliate_bindings (user_id, affiliate_id, tariff_kind) VALUES ($1, $2, $3)
                ON CONFLICT DO NOTHING`, [
                user_id,
                affiliate_id,
                tariff.tariff_kind
            ])

            if (result.rowCount) {
                info(`User #${user_id} was bound to affiliate #${affiliate_id} (${tariff.tariff_kind})`)
            } else {
                info(`User #${user_id} was not bound to affiliate #${affiliate_id} (${tariff.tariff_kind}) because he is bound to another one`)
            }


            return {status: codes.OK}

        })

    } catch (e) {
        error(`bindUserToAffiliate() err:`, e)
        return {status: codes.FAIL}
    }
}


/** accruing after the user's bet (using bet as a simple trigger to make the stretch out the accruing period) */
nats.subscribe('AFFILIATES.REVSHARE_ACCRUAL_TRIGGER', {queue: 'AFFILIATES.REVSHARE_ACCRUAL_TRIGGER'}, async ({user_id}, reply) => {

    const {status, accrued_usd_amount} = await performRevshareAccrual({user_id})

    if (accrued_usd_amount) {
        info(`$${accrued_usd_amount} was accrued to affiliate after the bet of user #${user_id}`)
    }

    nats.publish(reply, status)
})
